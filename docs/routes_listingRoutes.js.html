const express = require('express');
const router = express.Router();
const listingController = require('../controllers/listingController');
const { isAuthApi, isAuthPage } = require('../middleware/auth'); // Importa o middleware centralizado

/**
 * @fileoverview Rotas para gerenciar anúncios (listings) de cartas.
 * @requires express
 * @requires ../controllers/listingController
 * @requires ../middleware/auth
 */

/**
 * Rota para criar múltiplos anúncios de uma vez (em lote).
 * @name POST /api/listings/bulk-create
 * @function
 * @memberof module:routes/listingRoutes
 * @inner
 * @param {string} path - O caminho do endpoint.
 * @param {function} middleware - Middleware de autenticação da API.
 * @param {function} handler - Controller para processar a requisição.
 * @returns {object} 201 - Um objeto com a mensagem de sucesso e os dados dos anúncios criados.
 * @returns {object} 400 - Um erro de validação se os dados forem inválidos.
 * @returns {object} 500 - Um erro de servidor.
 */
router.post('/listings/bulk-create', isAuthApi, listingController.bulkCreateListings);

/**
 * Rota para exibir a página de edição de um anúncio.
 * @name GET /api/listings/:id/edit
 * @function
 * @memberof module:routes/listingRoutes
 * @inner
 * @param {string} path - O caminho do endpoint, com o ID do anúncio.
 * @param {function} middleware - Middleware de autenticação de página.
 * @param {function} handler - Controller para renderizar a página.
 * @param {string} req.params.id - O ID do anúncio a ser editado.
 * @returns {void} Renderiza a página de edição do anúncio.
 */
router.get('/listings/:id/edit', isAuthPage, listingController.showEditListingPage);

/**
 * Rota para atualizar um anúncio existente.
 * @name PUT /api/listings/:id
 * @function
 * @memberof module:routes/listingRoutes
 * @inner
 * @param {string} path - O caminho do endpoint, com o ID do anúncio.
 * @param {function} middleware - Middleware de autenticação de página.
 * @param {function} handler - Controller para processar a atualização.
 * @param {string} req.params.id - O ID do anúncio a ser atualizado.
 * @returns {object} 200 - Um objeto com a mensagem de sucesso.
 * @returns {object} 400 - Um erro de validação se os dados forem inválidos.
 * @returns {object} 404 - Se o anúncio não for encontrado.
 * @returns {object} 500 - Um erro de servidor.
 */
router.put('/listings/:id', isAuthPage, listingController.updateListing);

/**
 * Rota para deletar um anúncio.
 * @name DELETE /api/listings/:id
 * @function
 * @memberof module:routes/listingRoutes
 * @inner
 * @param {string} path - O caminho do endpoint, com o ID do anúncio.
 * @param {function} middleware - Middleware de autenticação de página.
 * @param {function} handler - Controller para processar a deleção.
 * @param {string} req.params.id - O ID do anúncio a ser deletado.
 * @returns {object} 200 - Um objeto com a mensagem de sucesso.
 * @returns {object} 404 - Se o anúncio não for encontrado.
 * @returns {object} 500 - Um erro de servidor.
 */
router.delete('/listings/:id', isAuthPage, listingController.deleteListing);

module.exports = router;
